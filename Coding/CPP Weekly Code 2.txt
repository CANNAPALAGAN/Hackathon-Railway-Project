class Solution {
public:
      int majorityElement(vector<int>& arr) {
 int n = arr.size();
    unordered_map<int, int> countMap;
    for (int num : arr) {
        countMap[num]++;
        if (countMap[num] > n / 2) {
            return num;
        }
    }
    return -1;
}

int main() {
    vector<int> arr = {1, 1, 2, 1, 3, 5, 1};
  
    cout << majorityElement(arr) << endl;

    return 0;     
    }
};
